152 /* A class that satisfies the Allocator concept, and which can therefore be
153  * used to manage memory for the standard C++ containers.
154  *
155  * For information about the various member functions, we refer to reference
156  * material for the Allocator concept, e.g.:
157  * http://en.cppreference.com/w/cpp/concept/Allocator
158  */
159 template<typename T>
160 class Allocator
161 {
162 public:
163  using value_type = T;
164 
165  explicit Allocator(const Memory& memory) : m_memory{memory} { }
166 
167  template<typename U>
168  Allocator(const Allocator<U>& other) CPPFMU_NOEXCEPT
169  : m_memory{other.m_memory}
170  {
171  }
172 
173  T* allocate(std::size_t n)
174  {
175  if (n == 0) return nullptr;
176  if (auto m = m_memory.Alloc(n, sizeof(T))) {
177  return reinterpret_cast<T*>(m);
178  } else {
179  throw std::bad_alloc();
180  }
181  }
182 
183  void deallocate(T* p, std::size_t n) CPPFMU_NOEXCEPT
184  {
185  if (n > 0) {
186  m_memory.Free(p);
187  }
188  }
189 
190  bool operator==(const Allocator& rhs) const CPPFMU_NOEXCEPT
191  {
192  return m_memory == rhs.m_memory;
193  }
194 
195  bool operator!=(const Allocator& rhs) const CPPFMU_NOEXCEPT
196  {
197  return !operator==(rhs);
198  }

201  // None of the following are required by C++11, yet they are, variously,
202  // required by GCC and MSVC.

204  template<typename U>
205  struct rebind { using other = Allocator<U>; };
206 
207 #if defined(__GNUC__) && (__GNUC__ < 5)
208  using pointer = T*;
209  using const_pointer = const T*;
210  using reference = T&;

212  using size_type = std::size_t;

256 /* Allocates memory for a single object of type T and runs its constructor,
257  * in the style of the built-in 'new' operator. Any arguments in 'args'
258  * are forwarded to the constructor.
259  */
260 template<typename T, typename... Args>
261 T* New(const Memory& memory, Args&&... args)
262 {
263  auto alloc = Allocator<T>{memory};
264  const auto ptr = std::allocator_traits<decltype(alloc)>::allocate(alloc, 1);
265  try {
266  std::allocator_traits<decltype(alloc)>::construct(
267  alloc,
268  ptr,
269  std::forward<Args>(args)...);
270  } catch (...) {
271  std::allocator_traits<decltype(alloc)>::deallocate(alloc, ptr, 1);
272  throw;
273  }
274  return ptr;
275 }

278 /* Destroys and deallocates memory for an object of type T, in the style of
279  * the built-in 'delete' operator.

281 template<typename T>
282 void Delete(const Memory& memory, T* obj) CPPFMU_NOEXCEPT
283 {

286  std::allocator_traits<decltype(alloc)>::deallocate(alloc, obj, 1);

290 /* An alias for a std::unique_ptr specialisation where the deleter is general
291  * and independent of the type of the object pointed to. This is used for the
292  * return type of AllocateUnique() below.
293  */
294 template<typename T>
295 using UniquePtr = std::unique_ptr<T, std::function<void(void*)>>;
296 
297 
298 /* Creates an object of type T which is managed by a std::unique_ptr.
299  * The object is created using cppfmu::New(), and when the time comes, it is
300  * destroyed using cppfmu::Delete().
301  */
302 template<typename T, typename... Args>
303 UniquePtr<T> AllocateUnique(const Memory& memory, Args&&... args)
304 {
305  return UniquePtr<T>{
306  New<T>(memory, std::forward<Args>(args)...),
307  [memory] (void* ptr) { Delete(memory, reinterpret_cast<T*>(ptr)); }};
308 }
